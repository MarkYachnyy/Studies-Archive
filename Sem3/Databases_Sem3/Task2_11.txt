--упр.2 (Напишите запрос с EXISTS и соединением, позволяющий вывести данные обо всех студентах обучающихся в вузах, имеющих рейтинг выше 300)
SELECT DISTINCT st.student_id, st.name, st.surname FROM student st, university un WHERE st.univ_id = un.univ_id AND un.rating > 300;
--упр.4 (Напишите запрос, выбирающий из таблицы SUBJECT данные о названиях предметов обучения, экзамены по которым сданы более чем одним студентом)
SELECT DISTINCT subj_name FROM subject su WHERE EXISTS 
(SELECT ex.subj_id FROM exam_marks ex WHERE ex.mark > 2 GROUP BY ex.subj_id HAVING ex.subj_id = su.subj_id AND COUNT(DISTINCT ex.student_id) > 1);
